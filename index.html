Evolution of websites

  • All of the aproaches have their pros & cons
  • All are in use today
  • All may be appropriate depending on the use case

Static websites

  • Simply styled HTML and CSS
  • Backend simply serves content doesn't do any logic
  • Good for hosting documents online (CVs, Books, Articles)

Generated websites

  • HTML is generated by the backend from templates and partials that contain logic
  • Forms submitted to the backend and state influence how the site displays
  • Simple and solid but can your have to refresh the page to get latest data

Asyncronous websites

  • HTML is served from the backend and modified on the client
  • Data is served asyncronously from the backend and updated on the client using js
  • Quickly gets messy, logic gets replicated on the server and client

Reactive websites

  • HTML (View) is generated on the client
  • HTML is re-rendered every time the state changes
  • State gets updated asyncronously

What is React

  • JavaScript library created by Facebook
  • User Interface (UI) library
  • Tool for building UI components

At it's core it's a library that takes some state and generates some HTML. It's all about components.

What is it good for

  • Organising code in complex websites into manageable chunks
  • Writing reusable components
  • Updating website as soon as new data arrives

import React, { useState } from "react";

function Placeholder(props) {
    const state = useState(true);
    const isCat = state[0];
    const setIsCat = state[1];
    let site = "";

    if (isCat) {
        site = "http://placekitten.com/g/";
    } else {
        site = "https://www.placecage.com/";
    }

    const toggleImage = function() {
        setIsCat(!isCat);
    };

    const url = site + props.width + "/" + props.height;

    return <img onClick={toggleImage} src={url} />;
}

export default Placeholder;

function Photoboard() {

        [{ id: 1, width: 10, height: 10 }, { id: 2, width: 10, height: 10 },  { id: 3, width: 10, height: 10 }]
    );
    const photos = state[0];
    const changePhotos = state[1];

    function grow() {
        const lastPhoto = photos[photos.length-1];
        const newObject = Object.assign({id: lastPhoto.id + 1}, lastPhoto])
        photos.push();
        photos.map(function(val, index) {
            photos[index] = { width: val.width * 2, height: val.height * 2 };
        });

        changePhotos(photos);
    }

    return (
        <div>
            {
                photos.map(
                    function(value) {
                        return <Placeholder key={value.id} width={value.width} height={value.height} />;
                    }
                )
            }
            <button onClick={grow}>Grow</button>
        </div>
    );
}

Exercise Three - Solution


import React, { useState } from "react";
import BoardPiece from "./BoardPiece";

import "./Board.css";

let turn = "Circle";

function Board() {
  const createdState = useState([0, 0, 0, 0, 0, 0, 0, 0, 0]);

  const boardState = createdState[0];
  const setBoardState = createdState[1];

  const onClick = function(index) {
    const newState = boardState.slice();
    newState[index] = turn;
    setBoardState(newState);
    if (turn === "Circle") {
      turn = "Cross";
    } else if (turn === "Cross") {
      turn = "Circle";
    }
  };

  return (
    <div className="Board">
      {boardState.map((elem, index) => {
        const onBoardPieceClick = function () {
            onClick(index)
        }

        return (
          <BoardPiece onClick={onBoardPieceClick} key={index} elem={elem} />
        );
      })}
    </div>
  );
}

export default Board;
